module reference.language.records;

type T := constructT : T;

type newtype :=
  mkNewtype@{
    f : T;
  };

type Pair (A B : Type) :=
  mkPair@{
    fst : A;
    snd : B;
  };

type EnumRecord :=
  | C1@{
      c1a : T;
      c1b : T;
    }
  | C2@{
      c2a : T;
      c2b : T;
    };

p1 : Pair T T :=
  mkPair@{
    fst := constructT;
    snd := constructT;
  };

p2 : Pair EnumRecord EnumRecord :=
  mkPair@{
    fst :=
      C1@{
        c1a := constructT;
        c1b := constructT;
      };
    snd :=
      C2@{
        c2a := constructT;
        c2b := constructT;
      };
  };

p1' : Pair T T :=
  mkPair@{
    fst := Pair.fst p1;
    snd := Pair.snd p1;
  };

module Open-Pair;
  open Pair;
  
  flipP : Pair T T := mkPair (snd p1) (fst p1);
end;

import Stdlib.Data.Nat open;

natPair : Pair Nat Nat :=
  mkPair@{
    fst := 1;
    snd := 2;
  };

updatedNatPair : Pair Nat Nat := natPair@Pair{fst := 2};

p3 : Pair Nat Nat :=
  let
    fst := 1;
    snd := 2;
  in mkPair@{
       fst := fst;
       snd := snd;
     };

p3' : Pair Nat Nat :=
  let
    fst := 1;
    snd := 2;
  in mkPair@{
       fst;
       snd;
     };

f : Pair Nat Nat -> Nat
  | mkPair@{fst := zero; snd} := snd
  | mkPair@{fst} := fst;
