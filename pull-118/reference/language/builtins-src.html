module reference.language.builtins;

import Stdlib.Data.Fixity open;

builtin nat
type Nat :=
  | zero : Nat
  | suc : Nat → Nat;

syntax operator + additive;
builtin nat-plus
+ : Nat → Nat → Nat
  | zero b := b
  | (suc a) b := suc (a + b);

module example-print-nat;
  
  builtin IO
  axiom IO : Type;

  builtin nat-print
  axiom printNat : Nat → IO;
end;
