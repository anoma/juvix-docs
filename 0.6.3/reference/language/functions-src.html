module reference.language.functions;

module example-multiply-by-two;
  import Stdlib.Data.Nat open using {Nat; *};
  
  multiplyByTwo (n : Nat) : Nat := 2 * n;
end;

module default-values;
  import Stdlib.Prelude open;
  
  f {x : Nat := 0} {y : Nat := 1} : Nat := x + y;
end;

module example-negate-boolean;
  import Stdlib.Data.Bool open;
  
  neg : Bool -> Bool
    | true := false
    | false := true;
end;

module move-to-left;
  import Stdlib.Data.Nat open;
  
  add (n : Nat) : Nat -> Nat
    | zero := n
    | (suc m) := suc (add n m);
end;

module example-add;
  import Stdlib.Data.Nat open;
  
  add : Nat -> Nat -> Nat
    | n zero := n
    | n (suc m) := suc (add n m);
end;

module short-definitons;
  import Stdlib.Data.Nat open;
  
  multiplyByTwo (n : Nat) : Nat := n;
end;

module mutually-recursive;
  import Stdlib.Data.Nat open;
  import Stdlib.Data.Bool open;
  import Stdlib.Prelude open;
  
  odd : Nat -> Bool
    | zero := false
    | (suc n) := even n;
  
  even : Nat -> Bool
    | zero := true
    | (suc n) := odd n;
end;

module anonymous-functions;
  import Stdlib.Prelude open;
  
  odd : Nat -> Bool :=
    \ {
      | zero := false
      | (suc n) := even n
    };
  
  even : Nat -> Bool :=
    \ {
      | zero := true
      | (suc n) := odd n
    };
end;
