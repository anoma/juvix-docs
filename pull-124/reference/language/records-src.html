module reference.language.records;

type T := constructT : T;

type newtype := mkNewtype@{f : T};

type Pair (A B : Type) :=
  mkPair@{
    fst : A;
    snd : B;
  };

p1 : Pair T T :=
  mkPair@{
    fst := constructT;
    snd := constructT;
  };

p1' : Pair T T :=
  mkPair@{
    fst := Pair.fst p1;
    snd := Pair.snd p1;
  };

p1'' : Pair T T :=
  mkPair@{
    fst := Pair.fst p1;
    snd := Pair.snd p1;
  };

open Pair;

flipP : Pair T T := mkPair (snd p1) (fst p1);

import Stdlib.Data.Nat open;

intPair : Pair Nat Nat :=
  mkPair@{
    fst := 1;
    snd := 2;
  };

updateIntPair : Pair Nat Nat := intPair@Pair{fst := 2};

type EnumRecord :=
  | C1@{
      c1a : T;
      c1b : T;
    }
  | C2@{
      c2a : T;
      c2b : T;
    };

p2 : Pair EnumRecord EnumRecord :=
  mkPair@{
    fst :=
      C1@{
        c1a := constructT;
        c1b := constructT;
      };
    snd :=
      C2@{
        c2a := constructT;
        c2b := constructT;
      };
  };
