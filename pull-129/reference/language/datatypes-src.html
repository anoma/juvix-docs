module reference.language.datatypes;

import Stdlib.Data.Fixity open;

type Bool :=
  | true : Bool
  | false : Bool;

type Unit := unit : Unit;

type Nat :=
  | zero : Nat
  | suc : Nat -> Nat;

syntax operator + additive;
+ : Nat -> Nat -> Nat
  | zero b := b
  | (suc a) b := suc (a + b);

module Nat-ADT;
  type Nat :=
    | Z
    | S Nat;
end;

module List-ADT;
  type List A :=
    | Nil
    | Cons A (List A);
end;

syntax operator :: cons;
type List (A : Type) :=
  | nil : List A
  | :: : A -> List A -> List A;

module membership;
  import Stdlib.Data.Bool open using {Bool; false; ||};
  
  isElem {A} (eq : A -> A -> Bool) (s : A) : List A -> Bool
    | nil := false
    | (x :: xs) := eq s x || isElem eq s xs;
end;
