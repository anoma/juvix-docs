module reference.language.pragmas;

import Stdlib.Prelude open using {Nat; zero; suc; List; ::; nil; module List};

{-# inline: true #-}
f : Nat -> Nat
  | x := x;

{-# inline: true, unroll: 100 #-}
g : Nat -> Nat
  | x := x;

axiom <body-f> : Nat -> Nat;

axiom <body-g> : Nat -> Nat;

axiom <body-h> : Nat -> Nat;

{-# inline: true #-}
module M;
  f : Nat -> Nat := <body-f>;
  
  g : Nat -> Nat := <body-g>;

  {-# inline: false #-}
  h : Nat -> Nat := <body-h>;
end;

{-# inline: 2 #-}
compose {A B C} (f : B -> C) (g : A -> B) (x : A) : C := f (g x);

{-# specialize: [f] #-}
map {A B} (f : A -> B) : List A -> List B
  | nil := nil
  | (x :: xs) := f x :: map f xs;

syntax alias T := Nat;
syntax alias T' := Nat;

map_g : List T -> List T'
  | nil := nil
  | (x :: xs) := g x :: map_g xs;

{-# inline: true #-}
funa {A} (f : A -> A) (a : A) : A :=
  let
    {-# specialize-by: [f] #-}
    go : Nat -> A
      | zero := a
      | (suc n) := f (go n);
  in go 10;

{-# specialize: true #-}
trait
type Natural N :=
  mkNatural@{
    + : N -> N -> N;
    * : N -> N -> N;
    fromNat : Nat -> N;
  };

axiom <body> : Natural Nat;

module pragma-specialise-instance;
  {-# specialize: true #-}
  instance
  naturalNatI : Natural Nat := <body>;
end;

module pragma-specialise-instance-false;
  {-# specialize: false #-}
  naturalNatI : Natural Nat := <body>;
end;
