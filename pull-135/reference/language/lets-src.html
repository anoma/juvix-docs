module reference.language.lets;

import Stdlib.Prelude open;

value : Nat :=
  let
    x : Nat := 5;
  in x * x;

value' : Nat :=
  let
    x := 5;
  in x * x;

sum (lst : List Nat) : Nat :=
  let
    go (acc : Nat) : List Nat -> Nat
      | nil := acc
      | (x :: xs) := go (acc + x) xs;
  in go 0 lst;

isEven (n : Nat) : Bool :=
  let
    isEven' : Nat -> Bool
      | zero := true
      | (suc n) := isOdd' n;
    isOdd' : Nat -> Bool
      | zero := false
      | (suc n) := isEven' n;
  in isEven' n;
